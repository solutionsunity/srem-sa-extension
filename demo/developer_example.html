üìã Basic Integration

Minimal code to integrate SREM bridge into your application:

// 1. Listen for extension responses window.addEventListener("message", (event) => { const data = event.data; if (data.type === "SREM_EXTENSION_PONG") { console.log("Extension is available"); } if (data.type === "SREM_AUTH_STATUS_RESPONSE") { console.log("Authenticated:", data.authenticated); } if (data.type === "SREM_BRIDGE_RESPONSE") { console.log("Search results:", data.result); } }); // 2. Check if extension exists (Public API) function checkExtension() { window.postMessage({ type: "SREM_EXTENSION_DISCOVERY" }, "*"); } // 3. Request domain approval (Public API) function requestApproval(appName, reason) { window.postMessage({ type: "SREM_REQUEST_APPROVAL", appName: appName, reason: reason }, "*"); } // 4. Check authentication status function checkAuth() { window.postMessage({ type: "SREM_AUTH_STATUS_REQUEST", requestId: "auth_" + Date.now() }, "*"); } // 5. Search deeds by owner function searchByOwner(deedNumbers, ownerIdType, ownerId) { window.postMessage({ type: "SREM_BRIDGE_REQUEST", requestId: "search_" + Date.now(), deedNumbers: deedNumbers, searchMode: "owner", ownerIdType: ownerIdType, ownerId: ownerId }, "*"); } // 6. Search deeds by date function searchByDate(deedNumbers, year, month, day, isHijri = false) { window.postMessage({ type: "SREM_BRIDGE_REQUEST", requestId: "search_" + Date.now(), deedNumbers: deedNumbers, searchMode: "date", deedDateYear: year, deedDateMonth: month, deedDateDay: day, isHijriDate: isHijri }, "*"); }
